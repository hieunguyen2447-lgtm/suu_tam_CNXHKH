   const quizElement = document.getElementById('quiz');
    const resultElement = document.getElementById('result');

    // 1. Render danh sách câu hỏi ra màn hình (Giữ nguyên thứ tự gốc 100%)
    function loadQuiz() {
        quizData.forEach((q, index) => {
            const questionDiv = document.createElement('div');
            questionDiv.classList.add('question');
            
            // In trực tiếp câu hỏi gốc
            questionDiv.innerHTML = `<p>${q.question}</p>`; 
            
            q.options.forEach((opt, optIndex) => {
                questionDiv.innerHTML += `
                    <label class="option">
                        <input type="radio" name="question${index}" value="${optIndex}">
                        ${opt}
                    </label>
                `;
            });
            quizElement.appendChild(questionDiv);
        });
    }

    // 2. Xử lý logic khi nộp bài
    function submitQuiz() {
        let score = 0;
        let attempted = 0;
        const total = quizData.length;
        
        quizData.forEach((q, index) => {
            const options = document.querySelectorAll(`input[name="question${index}"]`);
            let isAnswered = false;
            
            options.forEach((radio, optIndex) => {
                const label = radio.parentElement;
                
                // Trả về mặc định trước khi chấm
                label.style.color = '#333';
                label.style.fontWeight = 'normal';
                label.style.textDecoration = 'none';

                // Luôn bôi xanh đáp án đúng
                if (optIndex === q.answer) {
                    label.style.color = '#27ae60';
                    label.style.fontWeight = 'bold';
                }
                
                // Xử lý khi người dùng có tích chọn
                if (radio.checked) {
                    isAnswered = true;
                    if (optIndex === q.answer) {
                        score++;
                    } else {
                        // Chọn sai thì bôi đỏ, gạch ngang
                        label.style.color = '#e74c3c';
                        label.style.fontWeight = 'bold';
                        label.style.textDecoration = 'line-through';
                    }
                }
            });

            if (isAnswered) attempted++;
        });
        
        // Hiển thị kết quả
        resultElement.style.display = 'block';
        resultElement.innerText = `Bạn đã làm: ${attempted} / ${total} câu.\nBạn trả lời đúng: ${score} / ${total} câu!`;
        resultElement.scrollIntoView({ behavior: 'smooth' });
    }

    // Chạy khi trang vừa tải xong
    loadQuiz();
